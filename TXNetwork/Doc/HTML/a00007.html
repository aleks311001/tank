#include "TXLib.h"
#define ANOTHER_PEER_IP "10.0.2.203"
struct Netcursor_t
{
int x, y;
int old_x, old_y;
int action;
};
void NetPaint(TX_SOCKET paint_peer);
int main()
{
txCreateWindow (1024, 768);
TX_SOCKET paint_peer = txCreateSocket (TX_SERVER, ANOTHER_PEER_IP, TX_STD_PORT);
if (txnAssert(paint_peer) == TXN_NOT_CREATED)
{
printf("BAD SOCKET!!!");
return -1;
}
NetPaint (paint_peer);
return 0;
}
void NetPaint(TX_SOCKET paint_peer)
{
Netcursor_t cursor = {100, 100, -1, -1, 0};
Netcursor_t peer_cursor = {100, 100, -1, -1, 0};
while (!GetAsyncKeyState(VK_ESCAPE))
{
if (peer_cursor.action == -1)
break;
txSetFillColor (TX_GREEN);
txSetColor (TX_GREEN, 10);
if (cursor.action == 1) txLine (cursor.old_x, cursor.old_y, cursor.x, cursor.y);
txSetFillColor (TX_RED);
txSetColor (TX_RED, 10);
if (peer_cursor.action == 1) txLine (peer_cursor.old_x, peer_cursor.old_y, peer_cursor.x, peer_cursor.y);
cursor.old_x = cursor.x;
cursor.old_y = cursor.y;
cursor.x = txMouseX();
cursor.y = txMouseY();
if (txMouseButtons() == 1)
cursor.action = 1;
else
cursor.action = 0;
txSendTo(paint_peer, (char*) &cursor , sizeof(cursor));
txRecvTo(paint_peer, (char*) &peer_cursor, sizeof(peer_cursor));
txSleep(10);
}
cursor.action = -1;
txSendTo(paint_peer, (char*) &cursor, sizeof(cursor));
}

#include "TXLib.h"
#include "TXNetwork.h"
#define IP "127.0.0.1"
struct Ball_t
{
int x, y;
int vx, vy;
};
void Life(Ball_t* ball);
int main()
{
txCreateWindow (1024, 768);
Ball_t ball = {100, 100, 2, 3};
TX_SOCKET ball_client = txCreateSocket (TX_SERVER, IP, TX_STD_PORT);
if (txnAssert(ball_client) == TXN_NOT_CREATED)
{
printf("BAD SOCKET!!!");
return -1;
}
while (true)
{
txSendTo(ball_client, (char*) &ball, sizeof(ball));
txSetFillColor(TX_GREEN);
txSetColor(TX_GREEN);
txCircle(ball.x, ball.y, 10);
Life(&ball);
txSleep(1);
txSetFillColor(TX_BLACK);
txClear();
}
return 0;
}
void Life(Ball_t* ball)
{
if (ball->x < 0)
{
ball->x = 0;
ball->vx = -ball->vx;
}
if (ball->y < 0)
{
ball->y = 0;
ball->vy = -ball->vy;
}
if (ball->x > 1024)
{
ball->x = 1024;
ball->vx = -ball->vx;
}
if (ball->y > 768)
{
ball->y = 768;
ball->vy = -ball->vy;
}
ball->y += ball->vy;
ball->x += ball->vx;
}
