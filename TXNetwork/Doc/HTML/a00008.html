//{===========================================================================
// $Copyright: (C) _RedMan_ (Vova Yrovsky) <vova@txlib.ru> $
//}===========================================================================
#include "TXNetwork.h"
#include "TXLib.h"
#define ANOTHER_PEER_IP "127.0.0.1"
struct Player_t
{
int x, y;
int action;
};
struct Ball_t
{
int x, y;
int vx, vy;
};
void Game (TX_SOCKET peer_sock);
void DrawPlayer (Player_t player, COLORREF color);
void BallCalc (Ball_t* ball);
int main()
{
txCreateWindow (1024, 768);
TX_SOCKET client = txCreateSocket (TX_CLIENT, ANOTHER_PEER_IP, TX_STD_PORT, TX_BLOCKED, false);
if (txnAssert(client) == TXN_NOT_CREATED)
{
printf("BAD SOCKET!!!");
return -1;
}
Game(client);
return 0;
}
void Game(TX_SOCKET peer_sock)
{
Player_t me = {512, 748, 0};
Player_t peer = {512, 20 , 0};
Ball_t ball = {512, 384, 2, 2};
int timer = 0;
char sync[5] = "";
while (!GetAsyncKeyState (VK_ESCAPE) && peer.action != -1)
{
if (GetAsyncKeyState (VK_LEFT) && me.x > 50)
me.x -= 5;
if (GetAsyncKeyState (VK_RIGHT) && me.x < 1019)
me.x += 5;
BallCalc(&ball);
DrawPlayer(me , TX_GREEN);
DrawPlayer(peer, TX_RED);
timer = GetTickCount();
txRecvTo(peer_sock, (char*) &peer, sizeof(peer));
txSendTo(peer_sock, (char*) &me , sizeof(me));
txRecvTo(peer_sock, (char*) &ball, sizeof(ball));
if (GetTickCount() - timer < 15) txSleep(0);
txSetFillColor(TX_BLACK);
txClear();
}
}
void DrawPlayer(Player_t player, COLORREF color)
{
txSetColor(color);
txSetFillColor(color);
txRectangle(player.x - 100, player.y - 10, player.x + 100, player.y + 10);
}
void BallCalc(Ball_t* ball)
{
txSetColor(TX_BLUE);
txSetFillColor(TX_BLUE);
txCircle (ball->x, ball->y, 20);
}

Example4_Server_Code

//{===========================================================================
// $Copyright: (C) _RedMan_ (Vova Yrovsky) <vova@txlib.ru> $
//}===========================================================================
#include "TXNetwork.h"
#include "TXLib.h"
#define ANOTHER_PEER_IP "127.0.0.1"
struct Player_t
{
int x, y;
int action;
};
struct Ball_t
{
int x, y;
int vx, vy;
};
void Game (TX_SOCKET peer_sock);
void DrawPlayer (Player_t player, COLORREF color);
void BallCalc (Ball_t* ball);
void BallInterPlayers(Ball_t* ball, Player_t player1, Player_t player2);
int main()
{
txCreateWindow (1024, 768);
TX_SOCKET client = txCreateSocket (TX_SERVER, ANOTHER_PEER_IP, TX_STD_PORT, TX_BLOCKED, false);
if (txnAssert(client) == TXN_NOT_CREATED)
{
printf("BAD SOCKET!!!");
return -1;
}
Game(client);
return 0;
}
void Game(TX_SOCKET peer_sock)
{
Player_t me = {512, 20 , 0};
Player_t peer = {512, 748, 0};
Ball_t ball = {512, 384, 4, 4};
int timer = 0;
while (!GetAsyncKeyState (VK_ESCAPE) && peer.action != -1)
{
if (GetAsyncKeyState (VK_LEFT) && me.x > 50)
me.x -= 5;
if (GetAsyncKeyState (VK_RIGHT) && me.x < 1019)
me.x += 5;
BallCalc(&ball);
DrawPlayer(me , TX_GREEN);
DrawPlayer(peer, TX_RED);
BallInterPlayers(&ball, peer, me);
timer = GetTickCount();
txSendTo(peer_sock, (char*) &me , sizeof(me));
txRecvTo(peer_sock, (char*) &peer, sizeof(peer));
txSendTo(peer_sock, (char*) &ball, sizeof(ball));
if (GetTickCount() - timer < 25) txSleep (1);
txSetFillColor(TX_BLACK);
txClear();
}
}
void DrawPlayer(Player_t player, COLORREF color)
{
txSetColor(color);
txSetFillColor(color);
txRectangle(player.x - 100, player.y - 10, player.x + 100, player.y + 10);
}
void BallCalc(Ball_t* ball)
{
txSetColor(TX_BLUE);
txSetFillColor(TX_BLUE);
txCircle (ball->x, ball->y, 20);
if (ball->x < 0)
{
ball->x = 0;
ball->vx = -ball->vx;
}
if (ball->x > 1024)
{
ball->x = 1024;
ball->vx = -ball->vx;
}
if (ball->y < 0)
{
ball->y = 0;
ball->vy = -ball->vy;
}
if (ball->y > 768)
{
ball->y = 768;
ball->vy = -ball->vy;
}
ball->x -= ball->vx;
ball->y -= ball->vy;
}
void BallInterPlayers(Ball_t* ball, Player_t player1, Player_t player2)
{
if (player1.x - 100 < ball->x && ball->x < player1.x + 100 && player1.y - 25 < ball->y)
ball->vy = -ball->vy;
if (player2.x - 100 < ball->x && ball->x < player2.x + 100 && player2.y + 25 > ball->y)
ball->vy = -ball->vy;
}
