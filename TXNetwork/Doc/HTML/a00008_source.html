327 bool TX_SOCKET :: _txInitHowServer(u_short port)
328 {
329 _sock_addr = {};
330 _sock_addr.sin_family = AF_INET;
331 _sock_addr.sin_port = htons(port);
332 
333 if (bind(_sock, (sockaddr* ) &_sock_addr, sizeof(_sock_addr)))
334  return false;
335 
336 if (listen(_sock, 0x10))
337  return false;
338 
339 int client_addr_size = sizeof(_client_addr);
340 
341 _client_sock_for_serv = accept(_sock, (sockaddr* ) &_client_addr, &client_addr_size);
342 if (_client_sock_for_serv == 0) return false;
343 
344 return true;
345 }
