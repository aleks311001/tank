Example5_Client_Code

//{===========================================================================
// $Copyright: (C) _RedMan_ (Vova Yrovsky) <vova@txlib.ru> $
//}===========================================================================
#include "TXNetwork.h"
#include "TXLib.h"
struct Ball_t
{
int x, y;
int vx, vy;
};
int main()
{
txCreateWindow (1024, 768);
Ball_t ball = {0, 0, 0, 0};
TX_SOCKET ball_server = txCreateSocket (TX_CLIENT, TX_BROADCAST, TX_STD_PORT, TX_BLOCKED, false);
if (txnAssert(ball_server) == TXN_NOT_CREATED)
{
printf("BAD SOCKET!!!");
return -1;
}
while (true)
{
txRecvTo(ball_server, (char*) &ball, sizeof(ball));
txSetFillColor(TX_RED);
txSetColor(TX_RED);
txCircle(ball.x, ball.y, 10);
txSleep(25);
txSetFillColor(TX_BLACK);
txClear();
}
return 0;
}

Example5_Server_Code

//{===========================================================================
// $Copyright: (C) _RedMan_ (Vova Yrovsky) <vova@txlib.ru> $
//}===========================================================================
#include "TXNetwork.h"
#include "TXLib.h"
struct Ball_t
{
int x, y;
int vx, vy;
};
void Life(Ball_t* ball);
int main()
{
txCreateWindow (1024, 768);
Ball_t ball = {100, 100, 2, 3};
TX_SOCKET ball_client = txCreateSocket (TX_SERVER, TX_BROADCAST, TX_STD_PORT, TX_BLOCKED, false);
if (txnAssert(ball_client) == TXN_NOT_CREATED)
{
printf("BAD SOCKET!!!");
return -1;
}
while (true)
{
txSendTo(ball_client, (char*) &ball, sizeof(ball));
txSetFillColor(TX_GREEN);
txSetColor(TX_GREEN);
txCircle(ball.x, ball.y, 10);
Life(&ball);
txSleep(25);
txSetFillColor(TX_BLACK);
txClear();
}
return 0;
}
void Life(Ball_t* ball)
{
if (ball->x < 0)
{
ball->x = 0;
ball->vx = -ball->vx;
}
if (ball->y < 0)
{
ball->y = 0;
ball->vy = -ball->vy;
}
if (ball->x > 1024)
{
ball->x = 1024;
ball->vx = -ball->vx;
}
if (ball->y > 768)
{
ball->y = 768;
ball->vy = -ball->vy;
}
ball->y += ball->vy;
ball->x += ball->vx;
}
